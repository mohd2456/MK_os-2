PATTERN: Debounce function
function debounce(fn, delay) {
    let timer;
    return function(...args) {
        clearTimeout(timer);
        timer = setTimeout(() => fn.apply(this, args), delay);
    };
}
const debouncedSearch = debounce(search, 300);

PATTERN: Throttle function
function throttle(fn, limit) {
    let inThrottle;
    return function(...args) {
        if (!inThrottle) {
            fn.apply(this, args);
            inThrottle = true;
            setTimeout(() => inThrottle = false, limit);
        }
    };
}

PATTERN: Observable pattern
class EventEmitter {
    constructor() { this.events = {}; }
    on(event, fn) { (this.events[event] ||= []).push(fn); }
    off(event, fn) { this.events[event] = this.events[event]?.filter(f => f !== fn); }
    emit(event, ...args) { this.events[event]?.forEach(fn => fn(...args)); }
}

PATTERN: React custom hook
function useLocalStorage(key, initialValue) {
    const [value, setValue] = useState(() => {
        const stored = localStorage.getItem(key);
        return stored ? JSON.parse(stored) : initialValue;
    });
    useEffect(() => {
        localStorage.setItem(key, JSON.stringify(value));
    }, [key, value]);
    return [value, setValue];
}

PATTERN: Curry function
const curry = (fn) => {
    const arity = fn.length;
    return function curried(...args) {
        if (args.length >= arity) return fn(...args);
        return (...more) => curried(...args, ...more);
    };
};
const add = curry((a, b, c) => a + b + c);
add(1)(2)(3); // 6

PATTERN: Deep clone
function deepClone(obj) {
    if (obj === null || typeof obj !== 'object') return obj;
    if (obj instanceof Date) return new Date(obj);
    if (obj instanceof Array) return obj.map(item => deepClone(item));
    const clone = {};
    for (const key of Object.keys(obj)) {
        clone[key] = deepClone(obj[key]);
    }
    return clone;
}

PATTERN: Memoization
function memoize(fn) {
    const cache = new Map();
    return function(...args) {
        const key = JSON.stringify(args);
        if (cache.has(key)) return cache.get(key);
        const result = fn.apply(this, args);
        cache.set(key, result);
        return result;
    };
}

PATTERN: Pub/Sub pattern
class PubSub {
    constructor() { this.topics = {}; }
    subscribe(topic, fn) {
        this.topics[topic] = this.topics[topic] || [];
        this.topics[topic].push(fn);
        return () => this.topics[topic] = this.topics[topic].filter(f => f !== fn);
    }
    publish(topic, data) {
        (this.topics[topic] || []).forEach(fn => fn(data));
    }
}

PATTERN: Linked list
class ListNode {
    constructor(val, next = null) { this.val = val; this.next = next; }
}

class LinkedList {
    constructor() { this.head = null; }
    prepend(val) { this.head = new ListNode(val, this.head); }
    find(val) {
        let node = this.head;
        while (node) { if (node.val === val) return node; node = node.next; }
        return null;
    }
}

PATTERN: Binary search
function binarySearch(arr, target) {
    let lo = 0, hi = arr.length - 1;
    while (lo <= hi) {
        const mid = Math.floor((lo + hi) / 2);
        if (arr[mid] === target) return mid;
        if (arr[mid] < target) lo = mid + 1;
        else hi = mid - 1;
    }
    return -1;
}

PATTERN: Flat nested array
function flatten(arr, depth = Infinity) {
    return arr.reduce((acc, val) => {
        if (Array.isArray(val) && depth > 0) {
            return acc.concat(flatten(val, depth - 1));
        }
        return acc.concat(val);
    }, []);
}
// Or simply: arr.flat(Infinity);

PATTERN: Mutex with async
class AsyncMutex {
    #queue = [];
    #locked = false;
    async acquire() {
        if (this.#locked) {
            await new Promise(resolve => this.#queue.push(resolve));
        }
        this.#locked = true;
    }
    release() {
        if (this.#queue.length > 0) {
            this.#queue.shift()();
        } else {
            this.#locked = false;
        }
    }
}
